    -- 기본 영양소 (per 100g 또는 1회 제공량)
    serving_size REAL,          -- g 또는 ml
    calories REAL,              -- kcal
    protein REAL,               -- g
    fat REAL,                   -- g
    carbohydrate REAL,          -- g
    fiber REAL,                 -- g
    sodium REAL,                -- mg

    -- 비타민

    vitamin_d REAL,             -- μg
    vitamin_b1 REAL,            -- mg (티아민)
    vitamin_b2 REAL,            -- mg (리보플라빈)
    vitamin_b6 REAL,            -- mg
    vitamin_b12 REAL,           -- μg
    folate REAL,                -- μg (엽산)

    -- 미네랄
    calcium REAL,               -- mg
    iron REAL,                  -- mg
    magnesium REAL,             -- mg
    phosphorus REAL,            -- mg
    potassium REAL,             -- mg
    zinc REAL,                  -- mg

    -- 추가 정보
    food_group TEXT,            -- 식품군 (곡물, 채소, 과일 등)
    is_processed BOOLEAN,       -- 가공식품 여부
    nutrition_grade TEXT,       -- A/B/C/D/E (영양 등급)

## 🧮 지표 계산 알고리즘

### 1. 개인 영양 균형 점수 계산

```python
class NutritionAnalyzer:
    """개인 영양 분석 엔진"""

    # 한국인 영양소 섭취기준 (성인 기준)
    DAILY_REQUIREMENTS = {
        'calories': {'male': 2500, 'female': 2000, 'senior': 2000},
        'protein': {'male': 65, 'female': 55, 'senior': 60},
        'carbs': {'min': 130, 'optimal': 300},
        'fiber': {'min': 25},
        'sodium': {'max': 2000},  # mg
        'calcium': {'min': 800},
        'iron': {'male': 10, 'female': 14, 'senior': 9},
        'vitamin_c': {'min': 100},
        'vitamin_d': {'min': 10},
    }

    def calculate_nutrition_score(self, customer_phone: str, days: int = 30) -> Dict:
        """영양 균형 점수 계산"""

        # 1. 기간 내 주문 데이터 수집
        orders = self.get_customer_orders(customer_phone, days)

        # 2. 구매한 제품의 영양 정보 집계
        nutrition_data = self.aggregate_nutrition(orders)

        # 3. 일평균 섭취량 계산
        daily_avg = self.calculate_daily_average(nutrition_data, days)

        # 4. 각 영양소별 점수 계산
        scores = {
            'protein': self._score_nutrient(
                daily_avg['protein'], 
                self.DAILY_REQUIREMENTS['protein']['senior'],
                optimal_range=(0.8, 1.2)
            ),
            'carbs': self._score_nutrient(
                daily_avg['carbs'],
                self.DAILY_REQUIREMENTS['carbs']['optimal'],
                optimal_range=(0.9, 1.1)
            ),
            'fiber': self._score_nutrient(
                daily_avg['fiber'],
                self.DAILY_REQUIREMENTS['fiber']['min'],
                is_minimum=True
            ),
            'sodium': self._score_nutrient(
                daily_avg['sodium'],
                self.DAILY_REQUIREMENTS['sodium']['max'],
                is_maximum=True
            ),
            # ... 기타 영양소
        }

        # 5. 식품군 다양성 점수
        diversity_score = self._calculate_food_diversity(orders)

        # 6. 가공식품 비율
        processed_ratio = self._calculate_processed_ratio(orders)
        processed_penalty = max(0, processed_ratio - 0.3) * 50  # 30% 초과 시 감점

        # 7. 종합 점수 계산
        nutrient_avg = sum(scores.values()) / len(scores)

        final_score = (
            nutrient_avg * 0.6 +           # 영양소 균형 60%
            diversity_score * 0.3 +        # 다양성 30%
            (100 - processed_penalty) * 0.1  # 가공식품 비율 10%
        )

        return {
            'nutrition_balance_score': final_score,
            'nutrient_scores': scores,
            'food_diversity_score': diversity_score,
            'daily_avg': daily_avg,
            'deficiency': self._find_deficiencies(scores),
            'excess': self._find_excess(scores),
            'recommendations': self._generate_recommendations(scores, orders)
        }

    def _score_nutrient(self, actual: float, target: float, 
                       optimal_range=(0.9, 1.1), 
                       is_minimum=False, 
                       is_maximum=False) -> float:
        """개별 영양소 점수 계산"""

        if actual == 0:
            return 0

        ratio = actual / target

        if is_minimum:
            # 최소 권장량 기준 (많을수록 좋음, 단 과다도 감점)
            if ratio < 0.5:
                return ratio * 100  # 50% 미만: 비례 점수
            elif ratio < 1.0:
                return 50 + (ratio - 0.5) * 100  # 50-100%: 50-100점
            elif ratio <= 1.5:
                return 100  # 100-150%: 만점
            else:
                return max(0, 100 - (ratio - 1.5) * 40)  # 150% 초과: 감점

        elif is_maximum:
            # 최대 권장량 기준 (적을수록 좋음)
            if ratio <= 0.8:
                return 100
            elif ratio <= 1.0:
                return 100 - (ratio - 0.8) * 250  # 80-100%: 100-50점
            else:
                return max(0, 50 - (ratio - 1.0) * 50)  # 100% 초과: 급격히 감점

        else:
            # 최적 범위 기준
            lower, upper = optimal_range
            if lower <= ratio <= upper:
                return 100
            elif ratio < lower:
                return max(0, ratio / lower * 100)
            else:
                return max(0, 100 - (ratio - upper) / upper * 100)

    def _calculate_food_diversity(self, orders: List[Dict]) -> float:
        """식품군 다양성 점수"""

        food_groups = {
            '곡물': 0, '채소': 0, '과일': 0,
            '육류': 0, '생선': 0, '유제품': 0,
            '콩류': 0, '견과류': 0
        }

        for order in orders:
            for item in order['items']:
                group = self._get_food_group(item['category'])
                if group in food_groups:
                    food_groups[group] += 1

        # 섭취한 식품군 수
        consumed_groups = sum(1 for count in food_groups.values() if count > 0)

        # 각 식품군의 균형도
        total_items = sum(food_groups.values())
        if total_items == 0:
            return 0

        ideal_ratio = 1.0 / len(food_groups)
        balance = 0
        for count in food_groups.values():
            actual_ratio = count / total_items
            balance += (1 - abs(actual_ratio - ideal_ratio))

        balance_score = (balance / len(food_groups)) * 100

        # 다양성 점수 = 식품군 수 (50%) + 균형도 (50%)
        diversity_score = (consumed_groups / len(food_groups) * 50) + (balance_score * 0.5)

        return diversity_score
```

### 2. 지역 식품사막화 지수 계산

```python
class FoodDesertAnalyzer:
    """지역 식품사막화 분석 엔진"""

    def calculate_food_desert_index(self, sido: str, sigungu: str, 
                                    dong: str, quarter: str) -> Dict:
        """종합 식품사막화 지수 계산"""

        # 1. 접근성 지수 (0-100)
        accessibility = self._calculate_accessibility_index(sido, sigungu, dong)

        # 2. 다양성 지수 (0-100)
        diversity = self._calculate_diversity_index(sido, sigungu, dong, quarter)

        # 3. 경제성 지수 (0-100)
        affordability = self._calculate_affordability_index(sido, sigungu, dong, quarter)

        # 4. 영양 지수 (0-100)
        nutrition = self._calculate_nutrition_index(sido, sigungu, dong, quarter)

        # 5. 취약계층 지수 (0-100, 높을수록 취약)
        vulnerability = self._calculate_vulnerability_index(sido, sigungu, dong)

        # 6. 종합 지수 계산
        # 취약계층이 많을수록 다른 지표의 가중치 증가
        vulnerability_weight = vulnerability / 100

        food_desert_score = (
            accessibility * (0.25 + vulnerability_weight * 0.1) +
            diversity * (0.20 + vulnerability_weight * 0.05) +
            affordability * (0.20 + vulnerability_weight * 0.05) +
            nutrition * (0.25 + vulnerability_weight * 0.1) +
            (100 - vulnerability) * 0.10  # 취약계층 적을수록 가산점
        )

        # 7. 심각도 분류
        if food_desert_score >= 80:
            severity = 'safe'
        elif food_desert_score >= 60:
            severity = 'warning'
        elif food_desert_score >= 40:
            severity = 'danger'
        else:
            severity = 'critical'

        return {
            'food_desert_score': food_desert_score,
            'severity_level': severity,
            'accessibility_index': accessibility,
            'diversity_index': diversity,
            'affordability_index': affordability,
            'nutrition_index': nutrition,
            'vulnerability_index': vulnerability,
            'component_scores': {
                'accessibility': accessibility,
                'diversity': diversity,
                'affordability': affordability,
                'nutrition': nutrition,
                'vulnerability': vulnerability
            }
        }

    def _calculate_accessibility_index(self, sido: str, sigungu: str, dong: str) -> float:
        """접근성 지수"""

        # 평균 식품점 거리 (가까울수록 높은 점수)
        avg_distance = self._get_avg_mart_distance(sido, sigungu, dong)
        distance_score = max(0, 100 - avg_distance * 20)  # 5km 이상이면 0점

        # 대중교통 접근성
        transport_score = self._get_transport_score(sido, sigungu, dong)

        # 배달 커버리지
        delivery_rate = self._get_delivery_coverage(sido, sigungu, dong)
        delivery_score = delivery_rate * 100

        # 가중 평균
        accessibility = (
            distance_score * 0.4 +
            transport_score * 0.3 +
            delivery_score * 0.3
        )

        return accessibility

    def _calculate_diversity_index(self, sido: str, sigungu: str, 
                                   dong: str, quarter: str) -> float:
        """다양성 지수"""

        # 이용 가능 제품 수
        product_count = self._get_available_products(sido, sigungu, dong)
        product_score = min(100, product_count / 200 * 100)  # 200개 이상이면 만점

        # 식품군 다양성
        category_diversity = self._get_category_diversity(sido, sigungu, dong, quarter)

        # 신선식품 비율
        fresh_ratio = self._get_fresh_food_ratio(sido, sigungu, dong, quarter)
        fresh_score = fresh_ratio * 100

        diversity = (
            product_score * 0.3 +
            category_diversity * 0.4 +
            fresh_score * 0.3
        )

        return diversity

    def _calculate_affordability_index(self, sido: str, sigungu: str,
                                       dong: str, quarter: str) -> float:
        """경제성 지수"""

        # 지역 평균 가격 vs 전국 평균
        local_price = self._get_avg_price(sido, sigungu, dong, quarter)
        national_price = self._get_national_avg_price(quarter)

        if national_price > 0:
            price_ratio = local_price / national_price
            price_score = max(0, 100 - abs(price_ratio - 1.0) * 100)
        else:
            price_score = 50

        # 배달비 부담 정도
        delivery_fee = self._get_avg_delivery_fee(sido, sigungu, dong)
        delivery_burden = delivery_fee / local_price if local_price > 0 else 0
        delivery_score = max(0, 100 - delivery_burden * 500)  # 20% 이상이면 0점

        affordability = (
            price_score * 0.6 +
            delivery_score * 0.4
        )

        return affordability

    def _calculate_nutrition_index(self, sido: str, sigungu: str,
                                   dong: str, quarter: str) -> float:
        """영양 지수"""

        # 지역 주민 평균 영양 점수
        customers = self._get_region_customers(sido, sigungu, dong)

        if not customers:
            return 50  # 데이터 없으면 중간 점수

        nutrition_scores = []
        risk_count = 0

        for customer in customers:
            score = self._get_customer_nutrition_score(customer, quarter)
            nutrition_scores.append(score)

            if score < 60:  # 60점 미만을 영양 위험군으로 정의
                risk_count += 1

        avg_score = sum(nutrition_scores) / len(nutrition_scores)
        risk_rate = risk_count / len(nutrition_scores)

        # 평균 점수 70%, 위험군 비율 30%
        nutrition_index = avg_score * 0.7 + (1 - risk_rate) * 100 * 0.3

        return nutrition_index

    def _calculate_vulnerability_index(self, sido: str, sigungu: str, dong: str) -> float:
        """취약계층 지수 (높을수록 취약)"""

        # 인구 통계 데이터
        demographics = self._get_demographics(sido, sigungu, dong)

        # 고령자 비율
        senior_ratio = demographics['senior_population'] / demographics['total_population']
        senior_score = senior_ratio * 100  # 높을수록 취약

        # 저소득층 비율 (추정)
        service_usage_rate = demographics['using_service_count'] / demographics['total_households']
        income_score = (1 - service_usage_rate) * 100  # 서비스 미사용이 많으면 취약

        # 디지털 활용도
        digital_score = self._get_digital_literacy(sido, sigungu, dong)
        digital_vulnerability = 100 - digital_score

        vulnerability = (
            senior_score * 0.4 +
            income_score * 0.3 +
            digital_vulnerability * 0.3
        )

        return vulnerability
```

### 1. 데이터 익명화

```python
class PrivacyProtector:
    """개인정보 보호 처리"""

    @staticmethod
    def anonymize_location(address: str) -> Dict:
        """주소를 동/읍/면 수준으로만 저장"""
        # "서울시 강남구 역삼동 123-45" 
        # → {"sido": "서울", "sigungu": "강남구", "dong": "역삼동"}

        parts = address.split()
        return {
            'sido': parts[0] if len(parts) > 0 else '',
            'sigungu': parts[1] if len(parts) > 1 else '',
            'dong': parts[2] if len(parts) > 2 else '',
            # 상세 주소는 저장하지 않음
        }

    @staticmethod
    def aggregate_minimum_threshold(region_data: List) -> Optional[Dict]:
        """최소 집계 단위 (k-익명성 보장)"""

        MIN_THRESHOLD = 10  # 최소 10명 이상일 때만 통계 공개

        if len(region_data) < MIN_THRESHOLD:
            return None  # 데이터 부족 - 공개하지 않음

        return _calculate_statistics(region_data)

    @staticmethod
    def mask_sensitive_data(data: Dict) -> Dict:
        """민감 정보 마스킹"""
        masked = data.copy()

        if 'customer_phone' in masked:
            masked['customer_phone'] = _mask_phone(masked['customer_phone'])

        if 'address' in masked:
            masked['address'] = _mask_address(masked['address'])

        return masked
```
