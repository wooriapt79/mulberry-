```python
# 공개 제안
suggestion = manager.create_suggestion(
    user_id="user@mastodon.social",
    product_name="인제 산나물 세트",
    description="신선한 산나물",
    category="agricultural",
    visibility=SuggestionVisibility.PUBLIC
)

# 투표
manager.vote(suggestion.suggestion_id, user_id, "up")

# 검토 (관리자)
manager.review_suggestion(
    suggestion.suggestion_id,
    reviewer_id="admin",
    approved=True
)

# 실행 (공동구매 시작)
manager.implement_suggestion(suggestion.suggestion_id, campaign_id)
```
